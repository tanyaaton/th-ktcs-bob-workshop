/**
 * Payment response record with factory methods.
 * Combines immutability with flexible construction.
 */
public record PaymentResponse(
    String transactionId,
    TransactionStatus status,
    String authorizationCode,
    BigDecimal amount,
    String currency,
    String cardNumberMasked,
    String message,
    LocalDateTime timestamp
) {
    // Factory method for successful payments
    public static PaymentResponse success(
            String transactionId,
            String authCode,
            BigDecimal amount,
            String currency) {
        return new PaymentResponse(
            transactionId,
            TransactionStatus.SUCCESS,
            authCode,
            amount,
            currency,
            null,
            "Payment processed successfully",
            LocalDateTime.now()
        );
    }

    // Factory method for failed payments
    public static PaymentResponse failure(
            String transactionId,
            String errorMessage) {
        return new PaymentResponse(
            transactionId,
            TransactionStatus.FAILED,
            null, null, null, null,
            errorMessage,
            LocalDateTime.now()
        );
    }
}
